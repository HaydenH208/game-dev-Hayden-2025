// ===== 1. for loop ===== // Classic counted loop for (let i = 0; i < 5; i++) { console.log("for:", i); } // ===== 2. while loop ===== // Runs while condition is true let j = 0; while (j < 5) { console.log("while:", j); j++; } // ===== 3. do...while loop ===== // Executes at least once before checking condition let k = 0; do { console.log("do...while:", k); k++; } while (k < 5); // ===== 4. for...of loop ===== // Loops through iterable objects (arrays, strings, etc.) const fruits = ["apple", "banana", "cherry"]; for (const fruit of fruits) { console.log("for...of:", fruit); } // ===== 5. for...in loop ===== // Loops through enumerable object properties const person = { name: "Alice", age: 25, city: "NY" }; for (const key in person) { console.log("for...in:", key, "=", person[key]); } // ===== 6. Array.forEach() ===== // Executes a function for each array element const nums = [10, 20, 30]; nums.forEach((num, index) => { console.log("forEach:", index, num); }); // ===== 7. map(), filter(), reduce() ===== // Not technically loops, but often used similarly const doubled = nums.map(n => n * 2); // [20, 40, 60] const evens = nums.filter(n => n % 20 === 0); // [20] const sum = nums.reduce((acc, n) => acc + n, 0); // 60 console.log("map/filter/reduce:", doubled, evens, sum); // ===== 8. for await...of ===== // Used with async iterables (requires async context) async function asyncLoop() { const asyncNums = [Promise.resolve(1), Promise.resolve(2)]; for await (const n of asyncNums) { console.log("for await...of:", n); } } asyncLoop();
